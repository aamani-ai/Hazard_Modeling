numpy

pyarrow
duckdb

rasterio          # read/write GeoTIFF, ArcGIS exportImage tiles, windowed reads
rioxarray         # raster <-> xarray (CRS-aware clip/reproject)
rasterstats       # zonal statistics within the asset-boundary polygon

# Event footprint geometry (vectorize MESH swaths → polygons; GeoParquet/GeoJSON)
geopandas
shapely
# (pyproj is pulled in by geopandas — used for equal-area km² via EPSG:5070)

# Pluvial ponding fraction f from 1 m lidar (flood/m1_catalog/pluvial — JD-FL-15)
scikit-image      # morphological depression-fill (reconstruction) → closed-depression f
tifffile          # read the clipped 1 m DEM GeoTIFF into numpy (rasterio failed to build in this env)
# NOTE: also needs the GDAL command-line tools (gdalwarp / gdalbuildvrt) on PATH — present in the conda env;
